# toast

import { Toast } from 'solar-vue'
Vue.component(Toast.name, Toast)
/*
or
Vue.use(Icon)
*/

//*.vue
this.$toast({
  type: 'loading',
  loadingType: 'circle',
  message: '加载中'
})

<template>
  <sv-toast v-model="value">很高兴看到你</sv-toast>
</template>
<script>
export default { components: { [Toast.name]: Toast } }
</script>
